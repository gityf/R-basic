#3.矩阵
    #创建3行，2列的矩阵，依次赋值10到15
    mx <- matrix(10:15, nrow = 3, ncol = 2)
    #创建一个向量
    y1 <- c(1,2,3,4,5,6)
    #将向量转成3行2列的矩阵
    dim(y1) <- c(3,2)
    #将两个矩阵行拼接
    rbind(mx,y1)
    #将两个矩阵列拼接
    cbind(mx,y1)
